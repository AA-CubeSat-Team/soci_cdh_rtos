22/*



23 * Implements for the built-in timer of ECOS.



24 *



25 * Under Windows, we use QueryPerformanceCounter to obtain the counter value.



26 *



27 * Under Unix systems, we use clock_gettime function from <time.h>



28 */



29



30



31#include "timer.h"



32



33#if PROFILING > 0



34



35#if (defined WIN32 || _WIN64)



36



37#include <windows.h>



38



39void tic(timer* t)



40{



41	QueryPerformanceFrequency((LARGE_INTEGER*)&t->freq);



42	QueryPerformanceCounter((LARGE_INTEGER*)&t->tic);



43}



44



45pfloat toc(timer* t)



46{



47	QueryPerformanceCounter((LARGE_INTEGER*)&t->toc);



48	return ((t->toc - t->tic) / (pfloat)t->freq);



49}



50



51



52#elif (defined __APPLE__)



53



54void tic(timer* t)



55{



56    // read current clock cycles 



57    t->tic = mach_absolute_time();



58}



59



60pfloat toc(timer* t)



61{



62



63    uint64_t duration; /* elapsed time in clock cycles*/



64    



65    t->toc = mach_absolute_time();



66    duration = t->toc - t->tic;



67    



68    // conversion from clock cycles to nanoseconds



69    mach_timebase_info(&(t->tinfo));



70    duration *= t->tinfo.numer;



71    duration /= t->tinfo.denom;



72



73    return (pfloat)duration / 1000000000;



74}



75



76



77#elif (defined __linux__)



78



79/* read current time */



80void tic(timer* t)



81{



82	clock_gettime(CLOCK_MONOTONIC, &t->tic);



83}



84



85



86/* return time passed since last call to tic on this timer */



87double toc(timer* t)



88{



89	struct timespec temp;



90    



91	clock_gettime(CLOCK_MONOTONIC, &t->toc);



92    



93	if ((t->toc.tv_nsec - t->tic.tv_nsec)<0) {



94		temp.tv_sec = t->toc.tv_sec - t->tic.tv_sec-1;



95		temp.tv_nsec = 1000000000+t->toc.tv_nsec - t->tic.tv_nsec;



96	} else {



97		temp.tv_sec = t->toc.tv_sec - t->tic.tv_sec;



98		temp.tv_nsec = t->toc.tv_nsec - t->tic.tv_nsec;



99	}



100	return (pfloat)temp.tv_sec + (pfloat)temp.tv_nsec / 1000000000;



101}
