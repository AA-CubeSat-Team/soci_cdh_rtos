   63   /*
   64    * Builds KKT matrix.
   65    * We store and operate only on the upper triangular part.
   66    * Replace by or use in codegen.
   67    *
   68    * INPUT:      spmat* Gt - pointer to G'
   69    *             spmat* At - pointer to A'
   70    *               cone* C - pointer to cone struct
   71    *
   72    * OUTPUT:  idxint* Sign - pointer to vector of signs for regularization
   73    *              spmat* K - pointer to unpermuted upper triangular part of KKT matrix
   74    *         idxint* AttoK - vector of indices such that K[AtoK[i]] = A[i]
   75    *         idxint* GttoK - vector of indices such that K[GtoK[i]] = G[i]
   76    */
   77   void createKKT_U(spmat* Gt, spmat* At, cone* C, idxint** S, spmat** K,
   78                    idxint* AttoK, idxint* GttoK)
   79   {
   80   	idxint i, j, k, l, r, row_stop, row, cone_strt, ks, conesize;
   81   	idxint n = Gt->m;
   82   	idxint m = Gt->n;
   83   	idxint p = At ? At->n : 0;
   84   	idxint nK, nnzK;
   85   	pfloat *Kpr = NULL;
   86   	idxint *Kjc = NULL, *Kir = NULL;
   87       idxint *Sign;
   88   #ifdef EXPCONE
   89   	idxint exp_cone_strt;
   90   #endif
   91   
   92   	/* Dimension of KKT matrix
   93        *   =   n (number of variables)
   94        *     + p (number of equality constraints)
   95        *     + m (number of inequality constraints)
   96        *     + 2*C->nsoc (expansion of SOC scalings)
   97        */
   98       nK = n + p + m;
   99   #if CONEMODE == 0
  100       nK += 2*C->nsoc;
  101   #endif
  102   
  103       /* Number of non-zeros in KKT matrix
  104        *   =   At->nnz (nnz of equality constraint matrix A)
  105        *     + Gt->nnz (nnz of inequality constraint matrix)
  106        *     + C->lpc.p (nnz of LP cone)
  107        *     + 3*[sum(C->soc[i].p)+1] (nnz of expanded soc scalings)
  108        *     + 6*C->nexc (3x3 hessian of the exponential cone)
  109        */
  110   	nnzK = (At ? At->nnz : 0) + Gt->nnz + C->lpc->p;
  111   #if STATICREG == 1
  112       nnzK += n+p;
  113   #endif
  114   	for( i=0; i<C->nsoc; i++ ){
  115   #if CONEMODE == 0
  116   		nnzK += 3*C->soc[i].p+1;
  117   #else
  118           nnzK += (C->soc[i].p*(C->soc[i].p+1))/2;
  119   #endif
  120   	}
  121   
  122   #ifdef EXPCONE
  123       nnzK += 6*C->nexc;
  124   #endif
  125   
  126   
  127   #if PRINTLEVEL > 2
  128       PRINTTEXT("Non-zeros in KKT matrix: %d\n", (int) nnzK);
  129   #endif
  130   
  131   	/* Allocate memory for KKT matrix */
  132   	Kpr = (pfloat *)MALLOC(nnzK*sizeof(pfloat));
  133   	Kir = (idxint *)MALLOC(nnzK*sizeof(idxint));
  134   	Kjc = (idxint *)MALLOC((nK+1)*sizeof(idxint));
  135   
  136       /* Allocate memory for sign vector */
  137       Sign = (idxint *)MALLOC(nK*sizeof(idxint));
  138   #if PRINTLEVEL > 2
  139       PRINTTEXT("Memory allocated for sign vector\n");
  140   #endif
  141   
  142   	/* Set signs for regularization of (1,1) block */
  143       for( ks=0; ks < n; ks++ ){
  144           Sign[ks] = +1; /* (1,1) block */
  145       }
  146       for( ks=n; ks < n+p; ks++){
  147           Sign[ks] = -1; /* (2,2) block */
  148       }
  149   #if CONEMODE == 0
  150       for( ks=n+p; ks < n+p+C->lpc->p; ks++){
  151           Sign[ks] = -1; /* (3,3) block: LP cone */
  152       }
  153       ks = n+p+C->lpc->p;
  154       for( l=0; l<C->nsoc; l++){
  155           for (i=0; i<C->soc[l].p; i++) {
  156               Sign[ks++] = -1; /* (3,3) block: SOC, D */
  157           }
  158           Sign[ks++] = -1;     /* (3,3) block: SOC, v */
  159           Sign[ks++] = +1;     /* (3,3) block: SOC, u */
  160       }
  161   #else
  162       for (ks=n+p; ks < n+p+m; ks++) {
  163           Sign[ks] = -1;      /* (3,3) block has -1 sign if all dense */
  164       }
  165   #endif
  166   #ifdef EXPCONE /* In the case CONEMODE != 0 this is marginally inneficient */
  167       /* Set the signs of the regularization of the expcone block */
  168       for(ks=nK-3*C->nexc;ks<nK;)
  169       {
  170           Sign[ks++] = -1;
  171       }
  172   #endif
  173   
  174   
  175   #if DEBUG > 0
  176       if (ks!=nK) {
  177           PRINTTEXT("ks = %d, whereas nK = %d - exiting.", (int)ks, (int)nK);
  178           exit(-1);
  179       }
  180   #endif
  181   
  182       /* count the number of non-zero entries in K */
  183       k = 0;
  184   
  185       /* (1,1) block: the first n columns are empty */
  186   #if STATICREG == 0
  187       for (j=0; j<n; j++) {
  188           Kjc[j] = 0;
  189       }
  190   #else
  191       for (j=0; j<n; j++) {
  192           Kjc[j] = j;
  193           Kir[j] = j;
  194           Kpr[k++] = DELTASTAT;
  195       }
  196   #endif
  197   
  198       /* Fill upper triangular part of K with values */
  199       /* (1,2) block: A' */
  200   	i = 0; /* counter for non-zero entries in A or G, respectively */
  201   	for( j=0; j<p; j++ ){
  202           /* A' */
  203   		row = At->jc[j];
  204   		row_stop = At->jc[j+1];
  205   		if( row <= row_stop ){
  206   			Kjc[n+j] = k;
  207   			while( row++ < row_stop ){
  208   				Kir[k] = At->ir[i];
  209   				Kpr[k] = At->pr[i];
  210   				AttoK[i++] = k++;
  211   			}
  212   		}
  213   #if STATICREG == 1
  214           Kir[k] = n+j;
  215           Kpr[k++] = -DELTASTAT;
  216   #endif
  217       }
  218   	/* (1,3) and (3,3) block: [G'; 0; -Vinit]
  219        * where
  220        *
  221        *   Vinit = blkdiag(I, blkdiag(I,1,-1), ...,  blkdiag(I,1,-1));
  222        *                        ^ #number of second-order cones ^
  223        *
  224        * Note that we have to prepare the (3,3) block accordingly
  225        * (put zeros for init but store indices that are used in KKT_update
  226        * of cone module)
  227        */
  228   
  229   	/* LP cone */
  230   	i = 0;
  231   	for( j=0; j < C->lpc->p; j++ ){
  232           /* copy in G' */
  233   		row = Gt->jc[j];
  234   		row_stop = Gt->jc[j+1];
  235   		if( row <= row_stop ){
  236   			Kjc[n+p+j] = k;
  237   			while( row++ < row_stop ){
  238   				Kir[k] = Gt->ir[i];
  239   				Kpr[k] = Gt->pr[i];
  240   				GttoK[i++] = k++;
  241   			}
  242   		}
  243           /* -I for LP-cone */
  244   		C->lpc->kkt_idx[j] = k;
  245   		Kir[k] = n+p+j;
  246   		Kpr[k] = -1.0;
  247           k++;
  248   	}
  249   
  250   #if CONEMODE == 0
  251   	/* Second-order cones - copy in G' and set up the scaling matrix
  252        * which has the following structure:
  253        *
  254        *
  255        *    [ *             0  * ]
  256        *    [   *           *  * ]
  257        *    [     *         *  * ]       [ I   v  u  ]      I: identity of size conesize
  258        *  - [       *       *  * ]   =  -[ u'  1  0  ]      v: vector of size conesize - 1
  259        *    [         *     *  * ]       [ v'  0' -1 ]      u: vector of size conesize
  260        *    [           *   *  * ]
  261        *    [             * *  * ]
  262        *    [ 0 * * * * * * 1  0 ]
  263        *    [ * * * * * * * 0 -1 ]
  264        *
  265        * NOTE: only the upper triangular part (with the diagonal elements)
  266        *       is copied in here.
  267        */
  268   	cone_strt = C->lpc->p;
  269       for( l=0; l < C->nsoc; l++ ){
  270   
  271           /* size of the cone */
  272           conesize = C->soc[l].p;
  273   
  274           /* go column-wise about it */
  275   		for( j=0; j < conesize; j++ ){
  276   
  277              row = Gt->jc[cone_strt+j];
  278              row_stop = Gt->jc[cone_strt+j+1];
  279              if( row <= row_stop ){
  280                  Kjc[n+p+cone_strt+2*l+j] = k;
  281                  while( row++ < row_stop ){
  282                      Kir[k] = Gt->ir[i];
  283                      Kpr[k] = Gt->pr[i];
  284                      GttoK[i++] = k++;
  285                  }
  286              }
  287   
  288              /* diagonal D */
  289              Kir[k] = n+p+cone_strt + 2*l + j;
  290              Kpr[k] = -1.0;
  291              C->soc[l].Didx[j] = k;
  292              k++;
  293           }
  294   
  295           /* v */
  296           Kjc[n+p+cone_strt+2*l+conesize] = k;
  297           for (r=1; r<conesize; r++) {
  298               Kir[k] = n+p+cone_strt + 2*l + r;
  299               Kpr[k] = 0;
  300               k++;
  301           }
  302           Kir[k] = n+p+cone_strt + 2*l + conesize;
  303           Kpr[k] = -1;
  304           k++;
  305   
  306   
  307           /* u */
  308           Kjc[n+p+cone_strt+2*l+conesize+1] = k;
  309           for (r=0; r<conesize; r++) {
  310               Kir[k] = n+p+cone_strt + 2*l + r;
  311               Kpr[k] = 0;
  312               k++;
  313           }
  314           Kir[k] = n+p+cone_strt + 2*l + conesize + 1;
  315           Kpr[k] = +1;
  316           k++;
  317   
  318   
  319           /* prepare index for next cone */
  320   		cone_strt += C->soc[l].p;
  321   	}
  322   #else
  323       /* Second-order cones - copy in G' and set up the scaling matrix
  324        * which has a dense structure (only upper half part is shown):
  325        *
  326        *                     [ a        b*q'     ]
  327        *  - W^2 = -V = eta^2 [ b*q  I + c*(q*q') ]
  328        *
  329        * where    I: identity of size conesize
  330        *          q: vector of size consize - 1
  331        *      a,b,c: scalars
  332        *
  333        * NOTE: only the upper triangular part (with the diagonal elements)
  334        *       is copied in here.
  335        */
  336   	cone_strt = C->lpc->p;
  337       for( l=0; l < C->nsoc; l++ ){
  338   
  339           /* size of the cone */
  340           conesize = C->soc[l].p;
  341   
  342           /* go column-wise about it */
  343   		for( j=0; j < conesize; j++ ){
  344   
  345               row = Gt->jc[cone_strt+j];
  346               row_stop = Gt->jc[cone_strt+j+1];
  347               if( row <= row_stop ){
  348                   Kjc[n+p+cone_strt+j] = k;
  349                   while( row++ < row_stop ){
  350                       Kir[k] = Gt->ir[i];
  351                       Kpr[k] = Gt->pr[i];
  352                       GttoK[i++] = k++;
  353                   }
  354               }
  355   
  356               /* first elements - record where this column starts */
  357               Kir[k] = n+p+cone_strt;
  358               Kpr[k] = -1.0;
  359               C->soc[l].colstart[j] = k;
  360               k++;
  361   
  362               /* the rest of the column */
  363               for (r=1; r<=j; r++) {
  364                   Kir[k] = n+p+cone_strt+r;
  365                   Kpr[k] = -1.0;
  366                   k++;
  367               }
  368           }
  369   
  370           /* prepare index for next cone */
  371   		cone_strt += C->soc[l].p;
  372   	}
  373   #endif
  374   #ifdef EXPCONE
  375       /* Exponential cones - copy in G' and set up the scaling matrix
  376        * which has a dense structure (only upper half part is shown):
  377        *
  378        *  k is the index of the entry in Kip, Kir
  379        *  If the dense form of the socp cones is used the exponential cones start
  380        *  at row n+p+linear_vars+socp_vars.
  381        *  If the sparse form of the socp cones is used then the exponential cones will
  382        *  start at row n+p+linear_vars+socp_vars+2*number_socp_cones
  383        *  In either case cone_strt will have value equal to socp_vars+linear_vars.
  384        */
  385   
  386       /* At this point cone_strt = sum(q)+l */
  387   
  388       exp_cone_strt = cone_strt;
  389   #if CONEMODE == 0
  390       exp_cone_strt += 2*C->nsoc;
  391   #endif
  392        for( l=0; l < C->nexc; l++ ){
  393   
  394           /* go column-wise about it */
  395   		for( j=0; j < 3; j++ ){
  396               row = Gt->jc[cone_strt+j];
  397               row_stop = Gt->jc[cone_strt+j+1];
  398               if( row <= row_stop ){
  399                   Kjc[n+p+exp_cone_strt+j] = k;
  400                   while( row++ < row_stop ){
  401                       Kir[k] = Gt->ir[i];
  402                       Kpr[k] = Gt->pr[i];
  403                       GttoK[i++] = k++;
  404                   }
  405               }
  406   
  407               /* Save the index of the start of the column for the permutation */
  408               C->expc[l].colstart[j] = k;
  409               /* top of the column */
  410               for (r=0; r<j; r++) {
  411                   Kir[k] = n+p+exp_cone_strt+r;
  412                   Kpr[k] = 0.0;
  413                   k++;
  414               }
  415               /*diagonal*/
  416               Kir[k]   = n+p+exp_cone_strt+j;
  417               Kpr[k++] = -1.0;
  418           }
  419   
  420           /* prepare index for next cone */
  421   		cone_strt += 3;
  422           exp_cone_strt += 3;
  423   	}
  424   #endif
  425   
  426   
  427   #if PRINTLEVEL > 2
  428       PRINTTEXT("CREATEKKT: Written %d KKT entries\n", (int)k);
  429       PRINTTEXT("CREATEKKT: nK=%d and ks=%d\n",(int)nK,(int)ks);
  430       PRINTTEXT("CREATEKKT: Size of KKT matrix: %d\n", (int)nK);
  431   #endif
  432   
  433   	/* return Sign vector and KKT matrix */
  434       *S = Sign;
  435   	*K = ecoscreateSparseMatrix(nK, nK, nnzK, Kjc, Kir, Kpr);
  436   }

  438   /**
  439    * Cleanup: free memory (not used for embedded solvers, only standalone)
  440    *
  441    * Use the second argument to give the number of variables to NOT free.
  442    * This is useful if you want to use the result of the optimization without
  443    * copying over the arrays. One use case is the MEX interface, where we
  444    * do not want to free x,y,s,z (depending on the number of LHS).
  445    */
  446   void ECOS_cleanup(pwork* w, idxint keepvars)
  447   {
  448   	idxint i;
  449   
  450   #if defined EQUILIBRATE && EQUILIBRATE > 0
  451       /* unset equilibration */
  452       unset_equilibration(w);
  453   #endif
  454   
  455   	/* Free KKT related memory      ---            below are the corresponding MALLOCs                */
  456   	FREE(w->KKT->D);                /* mywork->KKT->D = (pfloat *)MALLOC(nK*sizeof(pfloat));          */
  457   	FREE(w->KKT->dx1);              /* mywork->KKT->dx1 = (pfloat *)MALLOC(mywork->n*sizeof(pfloat)); */
  458   	FREE(w->KKT->dx2);              /* mywork->KKT->dx2 = (pfloat *)MALLOC(mywork->n*sizeof(pfloat)); */
  459   	FREE(w->KKT->dy1);              /* mywork->KKT->dy1 = (pfloat *)MALLOC(mywork->p*sizeof(pfloat)); */
  460   	FREE(w->KKT->dy2);              /* mywork->KKT->dy2 = (pfloat *)MALLOC(mywork->p*sizeof(pfloat)); */
  461   	FREE(w->KKT->dz1);              /* mywork->KKT->dz1 = (pfloat *)MALLOC(mywork->m*sizeof(pfloat)); */
  462   	FREE(w->KKT->dz2);              /* mywork->KKT->dz2 = (pfloat *)MALLOC(mywork->m*sizeof(pfloat)); */
  463   	FREE(w->KKT->Flag);             /* mywork->KKT->Flag = (idxint *)MALLOC(nK*sizeof(idxint));       */
  464   	freeSparseMatrix(w->KKT->L);
  465   	FREE(w->KKT->Lnz);              /* mywork->KKT->Lnz = (idxint *)MALLOC(nK*sizeof(idxint));        */
  466   	FREE(w->KKT->Parent);           /* mywork->KKT->Parent = (idxint *)MALLOC(nK*sizeof(idxint));     */
  467   	FREE(w->KKT->Pattern);          /* mywork->KKT->Pattern = (idxint *)MALLOC(nK*sizeof(idxint));    */
  468   	FREE(w->KKT->Sign);             /* mywork->KKT->Sign = (idxint *)MALLOC(nK*sizeof(idxint));       */
  469   	FREE(w->KKT->Pinv);             /* mywork->KKT->Pinv = (idxint *)MALLOC(nK*sizeof(idxint));       */
  470       FREE(w->KKT->P);
  471   	FREE(w->KKT->PK);               /* mywork->KKT->PK = (idxint *)MALLOC(KU->nnz*sizeof(idxint));    */
  472   	freeSparseMatrix(w->KKT->PKPt); /* mywork->KKT->PKPt = newSparseMatrix(nK, nK, KU->nnz);          */
  473   	FREE(w->KKT->RHS1);             /* mywork->KKT->RHS1 = (pfloat *)MALLOC(nK*sizeof(pfloat));       */
  474   	FREE(w->KKT->RHS2);             /* mywork->KKT->RHS2 = (pfloat *)MALLOC(nK*sizeof(pfloat));       */
  475   	FREE(w->KKT->work1);            /* mywork->KKT->work1 = (pfloat *)MALLOC(nK*sizeof(pfloat));      */
  476   	FREE(w->KKT->work2);            /* mywork->KKT->work2 = (pfloat *)MALLOC(nK*sizeof(pfloat));      */
  477       FREE(w->KKT->work3);            /* mywork->KKT->work3 = (pfloat *)MALLOC(nK*sizeof(pfloat));      */
  478       FREE(w->KKT->work4);            /* mywork->KKT->work4 = (pfloat *)MALLOC(nK*sizeof(pfloat));      */
  479       FREE(w->KKT->work5);            /* mywork->KKT->work5 = (pfloat *)MALLOC(nK*sizeof(pfloat));      */
  480       FREE(w->KKT->work6);            /* mywork->KKT->work6 = (pfloat *)MALLOC(nK*sizeof(pfloat));      */
  481   	FREE(w->KKT);                   /* mywork->KKT = (kkt *)MALLOC(sizeof(kkt));                      */
  482   	if (w->A) {
  483   		FREE(w->AtoK);
  484   	}
  485   	FREE(w->GtoK);
  486   
  487   	/* Free memory for cones */
  488   	if( w->C->lpc->p > 0 ){
  489   		FREE(w->C->lpc->kkt_idx);
  490   		FREE(w->C->lpc->v);
  491   		FREE(w->C->lpc->w);
  492   	}
  493       /* C->lpc is always allocated, so we free it here. */
  494       FREE(w->C->lpc);
  495   
  496   	for( i=0; i < w->C->nsoc; i++ ){
  497   		FREE(w->C->soc[i].q);
  498   		FREE(w->C->soc[i].skbar);
  499   		FREE(w->C->soc[i].zkbar);
  500   #if CONEMODE == 0
  501   		FREE(w->C->soc[i].Didx);
  502   #endif
  503   #if CONEMODE > 0
  504           FREE(w->C->soc[i].colstart);
  505   #endif
  506   	}
  507   	if( w->C->nsoc > 0 ){
  508   		FREE(w->C->soc);
  509   	}
  510   #ifdef EXPCONE
  511       if(w->C->nexc > 0){
  512           FREE(w->C->expc);
  513       }
  514   #endif
  515   	FREE(w->C);
  516   
  517   	/* free stuff from pwork */
  518       FREE(w->W_times_dzaff);
  519   	FREE(w->dsaff_by_W);
  520       FREE(w->dzaff);
  521       FREE(w->dsaff);
  522       FREE(w->zaff);
  523       FREE(w->saff);
  524   	FREE(w->info);
  525       FREE(w->best_info);
  526   	FREE(w->lambda);
  527   	FREE(w->rx);
  528   	FREE(w->ry);
  529   	FREE(w->rz);
  530   	FREE(w->stgs);
  531   	FREE(w->G);
  532   	if( w->A ) FREE(w->A);
  533       FREE(w->best_z);
  534       FREE(w->best_s);
  535       FREE(w->best_y);
  536       FREE(w->best_x);
  537   	if( keepvars < 4 ) { FREE(w->z); }
  538   	if( keepvars < 3 ) { FREE(w->s); }
  539   	if( keepvars < 2 ) { FREE(w->y); }
  540   	if( keepvars < 1 ) { FREE(w->x); }
  541   #if defined EQUILIBRATE && EQUILIBRATE > 0
  542       FREE(w->xequil);
  543       FREE(w->Aequil);
  544       FREE(w->Gequil);
  545   #endif
  546   	FREE(w);
  547   }

  992   #endif

 1010   	mywork->rz = (m == 0) ? NULL : (pfloat *)MALLOC(m*sizeof(pfloat));
 1011
