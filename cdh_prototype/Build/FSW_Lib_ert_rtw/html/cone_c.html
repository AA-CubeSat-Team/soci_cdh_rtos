   33   /*
   34    * Computes u0^2 - u1'*u1 for second order cones
   35    */
   36   pfloat socres(pfloat* u, idxint p)
   37   {
   38   	pfloat res = u[0]*u[0];
   39   	idxint i;
   40   	for( i=1; i<p; i++) { res -= u[i]*u[i]; }
   41   	return res;
   42   }

   47   /**
   48    * Scales a conic variable such that it lies strictly in the cone.
   49    * If it is already in the cone, r is simply copied to s.
   50    * Otherwise s = r + (1+alpha)*e where alpha is the biggest residual.
   51    */
   52   void bring2cone(cone* C, pfloat* r, pfloat* s)
   53   {
   54   	pfloat alpha = -GAMMA;
   55   	pfloat cres, r1square;
   56   	idxint i, l, j;
   57   
   58   	/*
   59   	 * 1. Find maximum residual ----------------------------------------------
   60   	 */
   61   
   62   	/* LP cone */
   63   	for( i=0; i<C->lpc->p; i++ ){
   64   		if( r[i] <= 0 && -r[i] > alpha ){ alpha = -r[i]; }
   65   	}
   66   
   67   	/* Second-order cone */
   68   	for( l=0; l < C->nsoc; l++ ){
   69   		cres = r[i++]; r1square = 0;
   70   		for( j=1; j<C->soc[l].p; j++ ){ r1square += r[i]*r[i]; i++; }
   71   		cres -= sqrt(r1square);
   72   		if( cres <= 0 && -cres > alpha ){ alpha = -cres; }
   73   	}
   74   
   75   
   76   	/*
   77   	 * 2. compute s = r + (1+alpha)*e -------------------------------------
   78   	 */
   79   
   80   	alpha += 1.0;
   81   
   82   	/* LP cone */
   83   	for( i=0; i < C->lpc->p; i++ ){
   84   		s[i] = r[i] + alpha;
   85   	}
   86   
   87   	/* Second-order cone */
   88   	for( l=0; l < C->nsoc; l++ ){
   89   		s[i] = r[i] + alpha; i++;
   90   		for( j=1; j < C->soc[l].p; j++ ){ s[i] = r[i]; i++; }
   91   	}
   92   }

   96   /* Sets the initial point to the jordan algebra identity e times scaling for the symmetric cones
   97    * and the central ray for the exponential cone, scaled by scaling
   98    */
   99   void unitInitialization(cone* C, pfloat* s, pfloat* z, pfloat scaling)
  100   {
  101       idxint i,l,j;
  102   
  103       /* LP cone */
  104       for( i=0; i < C->lpc->p; i++ ){
  105           s[i] = scaling;
  106           z[i] = scaling;
  107       }
  108   
  109       /* Second-order cone */
  110       for( l=0; l < C->nsoc; l++ ){
  111           s[i] = scaling; z[i] = scaling;  i++;
  112           for( j=1; j < C->soc[l].p; j++ ){ s[i] = 0.0; z[i] = 0.0; i++; }
  113       }
  114       /* Exponential cone */
  115       for(l=0;l<C->nexc;l++)
  116       {
  117          s[i]   = scaling*(-1.051383945322714);
  118          s[i+1] = scaling*(1.258967884768947);
  119          s[i+2] = scaling*(0.556409619469370);
  120          z[i]   = scaling*(-1.051383945322714);
  121          z[i+1] = scaling*(1.258967884768947);
  122          z[i+2] = scaling*(0.556409619469370);
  123          i=i+3;
  124       }
  125   }

  130   /**
  131    * Update scalings.
  132    * Returns OUTSIDE_CONE as soon as any multiplier or slack leaves the cone,
  133    * as this indicates severe problems.
  134    */
  135   #ifdef EXPCONE
  136   idxint updateScalings(cone* C, pfloat* s, pfloat* z, pfloat* lambda, pfloat mu)
  137   #else
  138   idxint updateScalings(cone* C, pfloat* s, pfloat* z, pfloat* lambda)
  139   #endif
  140   {
  141   	idxint i, l, k, p; /*, pm1; */
  142   	pfloat sres, zres, snorm, znorm, gamma, one_over_2gamma;
  143   	pfloat* sk;
  144   	pfloat* zk;
  145       pfloat a, c, d, w, temp, divisor; /*, b; */
  146   #if CONEMODE == 0
  147       pfloat u0, u0_square, u1, v1, d1, c2byu02_d, c2byu02, c_square;
  148   #endif
  149   
  150   	/* LP cone */
  151   	for( i=0; i < C->lpc->p; i++ ){
  152   		C->lpc->v[i] = SAFEDIV_POS(s[i], z[i]);
  153   		C->lpc->w[i] = sqrt(C->lpc->v[i]);
  154   	}
  155   
  156   	/* Second-order cone */
  157   	k = C->lpc->p;
  158   	for( l=0; l < C->nsoc; l++ ){
  159   
  160   		/* indices and variables */
  161   		sk = s+k; zk = z+k; p = C->soc[l].p; /* pm1 = p-1; */
  162   
  163   		/* check residuals and quit if they're negative */
  164   		sres = socres(sk, p);  zres = socres(zk, p);
  165           if( sres <= 0 || zres <= 0 ){ return OUTSIDE_CONE; }
  166   
  167   		/* normalize variables */
  168   		snorm = sqrt(sres);    znorm = sqrt(zres);
  169   		for( i=0; i<p; i++ ){ C->soc[l].skbar[i] = SAFEDIV_POS(sk[i],snorm); }
  170   		for( i=0; i<p; i++ ){ C->soc[l].zkbar[i] = SAFEDIV_POS(zk[i],znorm); }
  171   		C->soc[l].eta_square = SAFEDIV_POS(snorm,znorm);
  172   		C->soc[l].eta = sqrt(C->soc[l].eta_square);
  173   
  174   		/* Normalized Nesterov-Todd scaling point */
  175   		gamma = 1.0;
  176   		for( i=0; i<p; i++){ gamma += C->soc[l].skbar[i]*C->soc[l].zkbar[i]; }
  177   		gamma = sqrt(0.5*gamma);
  178   		one_over_2gamma = SAFEDIV_POS(0.5,gamma);
  179   		a = one_over_2gamma*(C->soc[l].skbar[0] + C->soc[l].zkbar[0]);
  180   		w = 0;
  181   		for( i=1; i<p; i++ ){
  182   			C->soc[l].q[i-1] = one_over_2gamma*(C->soc[l].skbar[i] - C->soc[l].zkbar[i]);
  183   			w += C->soc[l].q[i-1]*C->soc[l].q[i-1];
  184   		}
  185           C->soc[l].w = w;
  186           C->soc[l].a = a;
  187   
  188   		/* pre-compute variables needed for KKT matrix (kkt_update uses those) */
  189           temp = 1.0 + a;
  190           /* b = SAFEDIV_POS(1.0,temp); */
  191           c = 1.0 + a + SAFEDIV_POS(w,temp);
  192           divisor = temp*temp;
  193           d = 1 + SAFEDIV_POS(2,temp) + SAFEDIV_POS(w,divisor);
  194   #if CONEMODE > 0
  195           C->soc[l].c = c;
  196           C->soc[l].d = d;
  197   #endif
  198   #if CONEMODE == 0
  199           c_square = c*c;
  200           divisor = 1.0 + w*d;
  201           d1 = 0.5*(a*a + w*(1.0 - SAFEDIV_POS(c_square,divisor) ));  if( d1 < 0 ){ d1 = 0; }
  202           u0_square = a*a + w - d1;
  203           u0 = sqrt(u0_square);
  204           c2byu02 = SAFEDIV_POS((c*c),u0_square);
  205           c2byu02_d = c2byu02 - d;
  206           if (c2byu02_d <= 0) { return OUTSIDE_CONE; }
  207           v1 = sqrt(c2byu02_d);
  208           u1 = sqrt(c2byu02);
  209           C->soc[l].d1 = d1;
  210           C->soc[l].u0 = u0;
  211           C->soc[l].u1 = u1;
  212           C->soc[l].v1 = v1;
  213   #endif
  214   
  215   		/* increase offset for next cone */
  216   		k += C->soc[l].p;
  217   	}
  218   #ifdef EXPCONE
  219          /*Exponential cones*/
  220           k = C->fexv;
  221           for(l=0;l<C->nexc;l++)
  222           {
  223               evalExpHessian(z+k, C->expc[l].v, mu);
  224               evalExpGradient(z+k, C->expc[l].g);
  225               k+=3;
  226           }
  227   #endif
  228   	/* lambda = W*z */
  229   	scale(z, C, lambda);
  230   
  231   	return INSIDE_CONE;
  232   }

  235   /* Evaulates log(s) + log(z) + log(t) + log(k) + logbarriersocp - (D+1) */
  236   pfloat evalSymmetricBarrierValue(pfloat* siter, pfloat *ziter, pfloat tauIter, pfloat kapIter, cone* C, pfloat D)
  237   {
  238      pfloat barrier = 0.0;
  239      idxint j,k,l;
  240      pfloat normAccumS = 0.0;
  241      pfloat normAccumZ = 0.0;
  242      idxint socDim;
  243      /* Positive orthant barrier */
  244      for(k=0;k<C->lpc->p;k++)
  245           barrier -= (siter[k]<=0||ziter[k]<=0)? ECOS_INFINITY : (log(siter[k])+log(ziter[k]));
  246   
  247       barrier-= (tauIter<=0||kapIter<=0) ? ECOS_INFINITY : (log(tauIter)+log(kapIter));
  248       /* Socp cones */
  249       for(l=0;l<C->nsoc;l++)
  250       {
  251           socDim = C->soc[l].p;
  252           normAccumS = 0.0;
  253           normAccumZ = 0.0;
  254           normAccumS = siter[k]*siter[k]; /* Root variable of the socp cone */
  255           normAccumZ = ziter[k]*ziter[k];
  256           k++;
  257           for(j=1;j<socDim;j++)
  258           {
  259               normAccumS -= siter[k]*siter[k];
  260               normAccumZ -= ziter[k]*ziter[k];
  261               k++;
  262           }
  263           barrier-= normAccumS<=0.0? ECOS_INFINITY : 0.5*log(normAccumS);
  264           barrier-= normAccumZ<=0.0? ECOS_INFINITY : 0.5*log(normAccumZ);
  265   
  266       }
  267       return barrier-D-1;
  268   }

  271   /**
  272    * Fast multiplication by scaling matrix.
  273    * Returns lambda = W*z
  274    * The exponential variables are not touched.
  275    */
  276   void scale(pfloat* z, cone* C, pfloat* lambda)
  277   {
  278   	idxint i, j, l, cone_start;
  279   	pfloat zeta, factor;
  280   
  281   	/* LP cone */
  282   	for( i=0; i < C->lpc->p; i++ ){ lambda[i] = C->lpc->w[i] * z[i]; }
  283   
  284   	/* Second-order cone */
  285   	cone_start = C->lpc->p;
  286   	for( l=0; l < C->nsoc; l++ ){
  287   
  288   		/* zeta = q'*z1 */
  289   		zeta = 0;
  290   		for( i=1; i < C->soc[l].p; i++ ){ zeta += C->soc[l].q[i-1] * z[cone_start + i]; }
  291   
  292   		/* factor = z0 + zeta / (1+a); */
  293   		factor = z[cone_start] + SAFEDIV_POS(zeta,(1+C->soc[l].a));
  294   
  295   		/* second pass (on k): write out result */
  296   		lambda[cone_start] = C->soc[l].eta*(C->soc[l].a*z[cone_start] + zeta); /* lambda[0] */
  297   		for( i=1; i < C->soc[l].p; i++ ){
  298   			j = cone_start+i;
  299   			lambda[j] = C->soc[l].eta*(z[j] + factor*C->soc[l].q[i-1]);
  300   		}
  301   
  302   		cone_start += C->soc[l].p;
  303   	}
  304   }

  307   /**
  308    *                                       [ D   v   u  ]
  309    * Fast multiplication with V =  eta^2 * [ v'  1   0  ] = W^2
  310    *                                       [ u   0  -1  ]
  311    * Computes y += W^2*x;
  312    */
  313   void scale2add(pfloat *x, pfloat* y, cone* C)
  314   {
  315       idxint i, l, cone_start, conesize, conesize_m1;
  316       pfloat *x1, *x2, *y1, *y2, eta_square, *q;
  317   #if CONEMODE == 0
  318       pfloat *x3, *x4, *y3, *y4;
  319       pfloat d1, u0, u1, v1;
  320       pfloat v1x3_plus_u1x4;
  321       pfloat qtx2;
  322   #else
  323       pfloat zeta, temp, a, w, c, d;
  324   #endif
  325   
  326       /* LP cone */
  327   	for( i=0; i < C->lpc->p; i++ ){ y[i] += C->lpc->v[i] * x[i]; }
  328   
  329       /* Second-order cone */
  330       cone_start = C->lpc->p;
  331   #if CONEMODE == 0
  332   
  333   	for( l=0; l < C->nsoc; l++ ){
  334   
  335           getSOCDetails(&C->soc[l], &conesize, &eta_square, &d1, &u0, &u1, &v1, &q);
  336           conesize_m1 = conesize - 1;
  337   
  338           x1 = x + cone_start;
  339           x2 = x1 + 1;
  340           x3 = x2 + conesize - 1;
  341           x4 = x3 + 1;
  342   
  343           y1 = y + cone_start;
  344           y2 = y1 + 1;
  345           y3 = y2 + conesize - 1;
  346           y4 = y3 + 1;
  347   
  348           /* y1 += d1*x1 + u0*x4 */
  349           y1[0] += eta_square*(d1*x1[0] + u0*x4[0]);
  350   
  351           /* y2 += x2 + v1*q*x3 + u1*q*x4 */
  352           v1x3_plus_u1x4 = v1*x3[0] + u1*x4[0];
  353           qtx2 = 0;
  354           for (i=0; i<conesize_m1; i++) {
  355               y2[i] += eta_square*(x2[i] + v1x3_plus_u1x4*q[i]);
  356               qtx2 += q[i]*x2[i];
  357           }
  358   
  359           /* y3 += v1*q'*x2 + x3 */
  360           y3[0] += eta_square*(v1*qtx2 + x3[0]);
  361   
  362           /* y4 += u0*x1 + u1*q'*x2 - x4 */
  363           y4[0] += eta_square*(u0*x1[0] + u1*qtx2 - x4[0]);
  364   
  365           /* prepare index for next cone */
  366           cone_start += conesize + 2;
  367       }
  368   
  369   #else
  370   
  371   	for( l=0; l < C->nsoc; l++ ){
  372   
  373           conesize = C->soc[l].p;
  374           conesize_m1 = conesize - 1;
  375           eta_square = C->soc[l].eta_square;
  376           a = C->soc[l].a;
  377           c = C->soc[l].c;
  378           d = C->soc[l].d;
  379           q = C->soc[l].q;
  380           w = C->soc[l].w;
  381   
  382           x1 = x + cone_start;
  383           x2 = x1 + 1;
  384   
  385           y1 = y + cone_start;
  386           y2 = y1 + 1;
  387   
  388           /* zeta = q'*x2 */
  389           zeta = 0;
  390           for (i=0; i<conesize-1; i++) {
  391               zeta += q[i]*x2[i];
  392           }
  393   
  394           /* y1 += eta^2*[ (a^2 + w)x1 + c*zeta ] */
  395           y1[0] += eta_square*( (a*a+w)*x1[0] + c*zeta );
  396   
  397           /* y2 += eta^2*[ (c*q*x1 + x2 + d*q*zeta ] */
  398           temp = c*x1[0] + d*zeta;
  399           for (i=0; i<conesize_m1; i++) {
  400               y2[i] += eta_square*( temp*q[i] + x2[i] );
  401           }
  402   
  403           /* prepare index for next cone */
  404           cone_start += conesize;
  405       }
  406   #endif
  407   #ifdef EXPCONE
  408       scaleToAddExpcone(y,x,C->expc,C->nexc,cone_start);
  409   #endif
  410   }

  412   /**
  413    * Fast left-division by scaling matrix.
  414    * Returns z = W\lambda
  415    */
  416   void unscale(pfloat* lambda, cone* C, pfloat* z)
  417   {
  418   	idxint i, j, l, cone_start;
  419   	pfloat zeta, factor;
  420   
  421   	/* LP cone */
  422   	for( i=0; i < C->lpc->p; i++ ){ z[i] = SAFEDIV_POS(lambda[i], C->lpc->w[i]); }
  423   
  424   	/* Second-order cone */
  425   	cone_start = C->lpc->p;
  426   	for( l=0; l < C->nsoc; l++ ){
  427   
  428   		/* zeta = q'*lambda1 */
  429   		zeta = 0;
  430   		for( i=1; i < C->soc[l].p; i++ ){ zeta += C->soc[l].q[i-1] * lambda[cone_start + i]; }
  431   
  432   		/* factor = -lambda0 + zeta / (1+a); */
  433   		factor = -lambda[cone_start] + SAFEDIV_POS(zeta,(1+C->soc[l].a));
  434   
  435   		/* second pass (on k): write out result */
  436   		z[cone_start] = SAFEDIV_POS( (C->soc[l].a*lambda[cone_start] - zeta), C->soc[l].eta );
  437   		for( i=1; i < C->soc[l].p; i++ ){
  438   			j = cone_start+i;
  439   			z[j] = SAFEDIV_POS( (lambda[j] + factor*C->soc[l].q[i-1]), C->soc[l].eta );
  440   		}
  441   
  442   		cone_start += C->soc[l].p;
  443   	}
  444   }

  448   /**
  449    * Conic product, implements the "o" operator, w = u o v
  450    * and returns e'*w (where e is the conic 1-vector)
  451    */
  452   pfloat conicProduct(pfloat* u, pfloat* v, cone* C, pfloat* w)
  453   {
  454   	idxint i, j, k, cone_start, conesize;
  455   	pfloat u0, v0, mu;
  456   
  457       mu = 0;
  458   	k=0;
  459   
  460   	/* LP cone */
  461   	for( i=0; i < C->lpc->p; i++ ){
  462           w[k] = u[i] * v[i];
  463           mu += w[k] < 0 ? -w[k] : w[k];
  464           k++;
  465       }
  466   
  467   	/* Second-order cone */
  468   	cone_start = C->lpc->p;
  469   	for( i=0; i < C->nsoc; i++ ){
  470   		conesize = C->soc[i].p;
  471   		u0 = u[cone_start];
  472   		v0 = v[cone_start];
  473   		w[k] = eddot(conesize, u+cone_start, v+cone_start);
  474           mu += w[k] < 0 ? -w[k] : w[k];
  475           k++;
  476   		for( j=1; j < conesize; j++ ){ w[k++] = u0*v[cone_start+j] + v0*u[cone_start+j]; }
  477   		cone_start += conesize;
  478   	}
  479   
  480       return mu;
  481   }

  485   /**
  486    * Conic division, implements the "\" operator, v = u \ w
  487    */
  488   void conicDivision(pfloat* u, pfloat* w, cone* C, pfloat* v)
  489   {
  490   	idxint i, j, k, cone_start, conesize;
  491   	pfloat rho, zeta, u0, w0, factor, temp;
  492   
  493   	/* LP cone */
  494   	for( i=0; i < C->lpc->p; i++ ){ v[i] = SAFEDIV_POS(w[i],u[i]); }
  495   
  496   	/* Second-order cone */
  497   	cone_start = C->lpc->p;
  498   	for( i=0; i < C->nsoc; i++ ){
  499   		conesize = C->soc[i].p;
  500   		u0 = u[cone_start]; w0 = w[cone_start];
  501   		rho = u0*u0;   zeta = 0;
  502   		for( j=1; j < conesize; j++ ){
  503   			k = cone_start+j;
  504   			rho -= u[k]*u[k];
  505   			zeta += u[k]*w[k];
  506   		}
  507           temp = SAFEDIV_POS(zeta,u0) - w0;
  508           factor = SAFEDIV_POS(temp,rho);
  509           temp = u0*w0 - zeta;
  510   		v[cone_start] = SAFEDIV_POS(temp,rho);
  511   		for( j=1; j < conesize; j++ ){
  512   			k = cone_start+j;
  513   			v[cone_start+j] = factor*u[k] + SAFEDIV_POS(w[k],u0);
  514   		}
  515   		cone_start += C->soc[i].p;
  516   	}
  517   }

  519   /*
  520    * Returns details on second order cone
  521    * Purpose: cleaner code
  522    */
  523   void getSOCDetails(socone *soc, idxint *conesize, pfloat* eta_square, pfloat* d1, pfloat* u0, pfloat* u1, pfloat* v1, pfloat **q)
  524   {
  525   #if CONEMODE == 0
  526       *conesize = soc->p;
  527       *eta_square = soc->eta_square;

  537   /*
  538    * Returns dx, dy and dz from the expanded and permuted version of
  539    * a search direction vector.
  540    */
  541   void unstretch(idxint n, idxint p, cone *C, idxint *Pinv, pfloat *Px, pfloat *dx, pfloat *dy, pfloat *dz)
  542   {
  543       idxint i,j,k,l;
  544       k = 0;
  545       for( i=0; i<n; i++ ){ dx[i] = Px[Pinv[k++]]; }
  546       for( i=0; i<p; i++ ){ dy[i] = Px[Pinv[k++]]; }
  547       j = 0;
  548       for( i=0; i<C->lpc->p; i++ ){ dz[j++] = Px[Pinv[k++]]; }
  549       for( l=0; l<C->nsoc; l++ ){
  550           for( i=0; i<C->soc[l].p; i++ ){ dz[j++] = Px[Pinv[k++]]; }
  551   #if CONEMODE == 0
  552           k += 2;
  553   #endif
  554       }
  555   #ifdef EXPCONE
  556       for( l=0; l<C->nexc; l++)
  557       {
  558           for( i=0; i<3; i++ ){ dz[j++] = Px[Pinv[k++]]; }
  559       }
  560   #endif
  561   }
