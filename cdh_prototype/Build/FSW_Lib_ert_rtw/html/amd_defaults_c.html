   17   /* ========================================================================= */
   18   /* === AMD defaults ======================================================== */
   19   /* ========================================================================= */
   20   
   21   GLOBAL void AMD_defaults
   22   (
   23       double Control [ ]
   24   )
   25   {
   26       Int i ;
   27   
   28       if (Control != (double *) NULL)
   29       {
   30   	for (i = 0 ; i < AMD_CONTROL ; i++)
   31   	{
   32   	    Control [i] = 0 ;
   33   	}
   34   	Control [AMD_DENSE] = AMD_DEFAULT_DENSE ;
   35   	Control [AMD_AGGRESSIVE] = AMD_DEFAULT_AGGRESSIVE ;
   36       }
   37   }
